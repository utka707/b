//jshint esversion:6 const express = require("express"); const bodyParser = require("body-parser"); var items=["Buy Food","Cook Food","Eat Food"]; const app = express(); app.use(bodyParser.urlencoded({extended: true})); app.set('view engine', 'ejs'); var a; app.get("/", function(req, res){ var currentday = new Date(); var options ={ weekday: "long", day: "numeric", month: "long", }; var day = currentday.toLocaleDateString("en-IN",options); res.render('list', {day: day,items:items}); }); app.post("/",function(req,res){ var item=req.body.newitem; items.push(item); res.redirect("/"); }); app.listen(3000, function(){ console.log("Server started on port 3000."); });
